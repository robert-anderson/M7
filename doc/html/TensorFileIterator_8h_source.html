70  FileIterator fi(filename, float_inds_regex);

107  static bool extract_line(const std::string &line, const size_t &nind,
108  const size_t &nreal_given, const bool &indsfirst, defs::inds &inds, T &value) {
109  ASSERT(inds.size() >= nind);
110  std::istringstream iss(line);
111  std::vector<std::string> split(std::istream_iterator<std::string>{iss},
112  std::istream_iterator<std::string>());
113  for (size_t i = 0ul; i < nind; ++i) {
114  inds[i] = atoi((split.begin() + (indsfirst ? 0 : 1) + i)->data()) - 1;
115  }
116 
117  std::smatch match;
118  std::regex_search(line, match, nreal_given == 1 ? float_regex : complex_regex);
119  if (!match.size()) return false;
120  std::istringstream(match.str()) >> value;
121  return true;
122  }

124  static defs::inds shape(const std::string &filename, const size_t &nind,
125  const size_t &nreal_given, const bool &indsfirst, const std::regex &data_regex) {
126  FileIterator fi(filename, data_regex);
127  std::string line;
128  defs::inds shape(nind, 0ul);
129  defs::inds inds(nind, 0ul);
130  T value;
131  while (fi.next(line)) {
132  extract_line(line, nind, nreal_given, indsfirst, inds, value);
133  for (size_t i = 0ul; i < nind; ++i) {
134  if (inds[i] < (size_t) (-1) && inds[i] + 1 > shape[i]) shape[i] = inds[i] + 1;
135  }
136  }
137  return shape;
138  }
