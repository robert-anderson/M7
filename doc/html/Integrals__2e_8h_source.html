26 template<typename T, size_t isym>
27 class Integrals_2e : public Integrals {
28  // none i ih ihr
29  static_assert(isym == 1 || isym == 2 || isym == 4 || isym == 8, "Invalid symmetry parameter specified.");
30  const size_t m_norb2, m_norb3, m_nelem_8fold, m_nelem;
31  std::vector<T> m_data;
32 public:
33  Integrals_2e(const size_t &norb, bool spin_resolved) :
34  Integrals(norb, spin_resolved), m_norb2(m_norb * norb), m_norb3(m_norb2 * norb),
35  m_nelem_8fold(trig(0, trig(0, norb))), m_nelem(nelem(norb)) {
36  m_data.resize(m_nelem, 0.0);

39  Integrals_2e(std::string fname, bool spin_major = false) :
40  Integrals_2e(FcidumpFileIterator<T>(fname).m_norb, FcidumpFileIterator<T>(fname).m_spin_resolved) {
41  FcidumpFileIterator<T> file_iterator(fname);
42  defs::inds inds(4);
43  T value;
44  while (file_iterator.next(inds, value)) {

51  /*
52  * given the indices and their identified case, return the flat indices
53  */
54  inline size_t flat_index(const size_t &icase, const size_t &i, const size_t &j,
55  const size_t &k, const size_t &l) const {
56  switch (icase) {
57  case 0:
58  return i + j * m_norb + k * m_norb2 + l * m_norb3;
59  case 1:
60  return trig(i + j * m_norb, k + l * m_norb);
61  case 2:
62  return trig(k + l * m_norb, i + j * m_norb);
63  case 3:
64  return trig(trig(i, j), trig(k, l));
65  case 4:
66  return trig(trig(k, l), trig(i, j));
67  case 5:
68  return trig(trig(i, j), trig(l, k)) + (isym == 4 ? m_nelem_8fold : 0);
69  case 6:
70  return trig(trig(l, k), trig(i, j)) + (isym == 4 ? m_nelem_8fold : 0);
71  case 7:
72  return trig(trig(j, i), trig(k, l)) + (isym == 4 ? m_nelem_8fold : 0);
73  case 8:
74  return trig(trig(k, l), trig(j, i)) + (isym == 4 ? m_nelem_8fold : 0);
75  case 9:
76  return trig(trig(j, i), trig(l, k));
77  case 10:
78  return trig(trig(l, k), trig(j, i));
79  }
80  }

82  void set(const size_t &i, const size_t &j, const size_t &k, const size_t &l, const T &value) {
83  auto icase = get_case(i, j, k, l);
84  auto iflat = flat_index(icase, i, j, k, l);
85  auto conjd_value = case_to_tconj[icase] ? consts::conj(value) : value;
86  if (consts::float_is_zero(m_data[iflat])) m_data[iflat] = conjd_value;
87  else {
88  ASSERT(consts::floats_nearly_equal(m_data[iflat], conjd_value));
89  }
90  }

106  void set_from_fcidump(const defs::inds &inds, const T &value, bool spin_major = false) {
107  /*
108  * spin-resolved FCIDUMPs index in spinorbs, which may not may not be spin-major,
109  * depending on the program they were generated for. E.g. NECI uses spatial-major
110  * ordering throughout, so if the FCIDUMP supplied was intended for use with NECI,
111  * spin_major should be passed in as false.
112  */
113  if (!m_spin_resolved) set(inds, value);
114  else if (spin_major)
115  set(inds, value);
116  else
117  set(
118  inds[0] / 2, inds[0] % 2, inds[1] / 2, inds[1] % 2,
119  inds[2] / 2, inds[2] % 2, inds[3] / 2, inds[3] % 2, value
120  );
121 
122  }

124  T get(const size_t &i, const size_t &j, const size_t &k, const size_t &l) const {
125  ASSERT(i < m_norb && j < m_norb && k < m_norb && l < m_norb);
126  auto icase = get_case(i, j, k, l);
127  auto iflat = flat_index(icase, i, j, k, l);
128  return case_to_tconj[icase] ? consts::conj(m_data[iflat]) : m_data[iflat];

131  T element(const size_t &i, const size_t &j, const size_t &k, const size_t &l) const {
132  /*
133  * return spin-orbital indexed integral from indices in chemists' ordering
134  */
135  if (m_spin_resolved) return get(i, j, k, l);
136  else {
137  // enforce spin symmetry
138  if (((i<m_norb)!=(j<m_norb)) || ((k<m_norb)!=(l<m_norb))) return 0.0;
139  return get(i%m_norb, j%m_norb, k%m_norb, l%m_norb);
140  }

186  constexpr size_t nelem(const size_t &norb) {
187  static_assert(isym == 1 || isym == 2 || isym == 4 || isym == 8, "Invalid symmetry parameter specified.");
188  switch (isym) {
189  case 1 :
190  return norb * norb * norb * norb;
191  case 2 :
192  return trig(0, norb * norb);
193  case 4 :
194  return 2 * trig(0, trig(0, norb));
195  case 8 :
196  return trig(0, trig(0, norb));
197  }
198  }

200  constexpr size_t get_case(const size_t &i, const size_t &j, const size_t &k, const size_t &l) const {
201  if (isym == 1) {
202  return 0;
203  } else if (isym == 2) {
204  /*
205  * Only assuming I symmetry
206  * rect(i, k) <= rect(j, l)
207  */
208  return (i < j || (i == j && k < l)) ? 1 : 2;
209  } else if (isym > 2) {
210  /*
211  * in IH and IHR symmetry, there are 8 identifiable cases
212  */
213  if (i <= j) {
214  if (k <= l) return (trig(i, j) <= trig(k, l)) ? 3 : 4;
215  else return (trig(i, j) <= trig(l, k)) ? 5 : 6;
216  } else {
217  if (k <= l) return (trig(j, i) <= trig(k, l)) ? 7 : 8;
218  else return (trig(j, i) <= trig(l, k)) ? 9 : 10;
219  }
220  }
221  }
