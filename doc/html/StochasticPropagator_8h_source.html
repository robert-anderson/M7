31  void off_diagonal(const DeterminantElement &src_det, const NumericElement<defs::ham_t> &weight,
32  SpawnList &spawn_list, bool flag_deterministic, bool flag_initiator) override {
33  ASSERT(!consts::float_is_zero(*weight));
34  m_occ.get().update(src_det);
35  m_vac.get().update(src_det);
36  size_t nattempt = std::ceil(std::abs(*weight));
37  defs::prob_t prob;
38  defs::ham_t helem;
39  bool valid;
40 
41  for (size_t iattempt = 0ul; iattempt < nattempt; ++iattempt) {
42  size_t nexcit = 2 - m_prng.get().stochastic_round(m_magnitude_logger.m_psingle, 1);
43  switch (nexcit) {
44  case 1:
45  valid = m_exgen->draw_single(src_det, m_dst_det.get(), m_occ.get(), m_vac.get(), prob, helem, m_anticonn.get());
46  prob*=m_magnitude_logger.m_psingle;
47  ASSERT(!consts::float_nearly_zero(prob, 1e-14));
48  break;
49  case 2:
50  // TODO: don't need m_vac for doubles.
51  valid = m_exgen->draw_double(src_det, m_dst_det.get(), m_occ.get(), prob, helem, m_anticonn.get());
52  prob*= 1.0-m_magnitude_logger.m_psingle;
53  break;
54  }
55  if (!valid) continue;
56 
57  ASSERT(!consts::float_is_zero(prob))
58  auto delta = -(*weight / (defs::ham_comp_t) nattempt) * m_tau * helem /prob;
59  delta = m_prng.get().stochastic_threshold(delta, m_min_spawn_mag);
60  if (consts::float_is_zero(delta)) continue;
61  spawn(spawn_list, m_dst_det.get(), delta, flag_initiator);
62  m_magnitude_logger.log(nexcit, helem, prob);
63  }
64  }

66  void diagonal(const NumericElement<defs::ham_comp_t> &hdiag, NumericElement<defs::ham_t> &weight,
67  defs::ham_comp_t &delta_square_norm, defs::ham_comp_t &delta_nw) override {
68 
69  delta_square_norm -= std::pow(std::abs(*weight), 2);
70  delta_nw -= std::abs(*weight);
71 
72  // the probability that each unit walker will die
73  auto death_rate = (*hdiag - m_shift) * m_tau;
74  ASSERT(std::abs(death_rate) < 1)
75 
76  weight = m_prng.get().stochastic_round(*weight, 1.0) * (1 - death_rate);
77 
78  delta_square_norm += std::pow(std::abs(*weight), 2);
79  delta_nw += std::abs(*weight);
80  }
