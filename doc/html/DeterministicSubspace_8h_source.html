54  void build_hamiltonian(const Hamiltonian *ham) {
55  ASSERT(m_sparse_ham.empty());
56  m_full_subspace_list->all_gather(*m_local_subspace_list);
57  m_sparse_ham.resize(nrow_local());
58 #pragma omp parallel for default(none) shared(ham)
59  for (size_t irow_local = 0ul; irow_local < nrow_local(); ++irow_local) {
60  // loop over local subspace (H rows)
61  auto row_det = m_local_subspace_list->m_determinant(irow_local);
62  for (size_t irow_full = 0ul; irow_full < nrow_full(); ++irow_full) {
63  // loop over full subspace (H columns)
64  // only add to sparse H if dets are connected
65  auto col_det = m_full_subspace_list->m_determinant(irow_full);
66  if (row_det == col_det) continue;
67  auto helem = ham->get_element(row_det, col_det);
68  if (!consts::float_is_zero(helem)) m_sparse_ham(irow_local, irow_full) = helem;
69  }
70  }

72  m_local_weights.resize(nrow_local(), 0.0);
73  m_local_h_weights.resize(nrow_local(), 0.0);
74  m_full_weights.resize(nrow_full(), 0.0);
75  size_t nrow = nrow_local();
76  mpi::all_gather(&nrow, 1, m_recvcounts.data(), 1);
77  for (size_t i = 1ul; i < mpi::nrank(); ++i) m_displs[i] = m_displs[i - 1] + m_recvcounts[i - 1];

83  void gather_and_project() {
84 #pragma omp parallel for default(none) shared(stderr)
85  for (size_t irow_local = 0ul; irow_local < nrow_local(); ++irow_local) {
86  auto irow_walker_list = *m_local_subspace_list->m_irow(irow_local);
87  ASSERT(m_walker_list->m_flags.m_deterministic(irow_walker_list));
88  m_local_weights[irow_local] = *m_walker_list->m_weight(irow_walker_list);
89  ASSERT(m_local_weights[irow_local]==m_local_weights[irow_local])
90  }
91  ASSERT(nrow_local()==m_recvcounts[mpi::irank()])
92  ASSERT(nrow_local()==m_local_weights.size())
93  mpi::all_gatherv(m_local_weights.data(), nrow_local(), m_full_weights.data(), m_recvcounts, m_displs);
94  m_local_h_weights.assign(m_local_h_weights.size(), 0);
95  m_sparse_ham.multiply(m_full_weights, m_local_h_weights);

98  void rayleigh_quotient(Hybrid<defs::ham_t> &num, Hybrid<defs::ham_comp_t> &norm_square){
99 #pragma omp parallel for default(none) shared(num, norm_square)
100  for (size_t irow_local = 0ul; irow_local < nrow_local(); ++irow_local) {
101  auto irow_walker_list = *m_local_subspace_list->m_irow(irow_local);
102  auto w = m_local_weights[irow_local];
103  auto hw = m_local_h_weights[irow_local];
104  num.thread()+=consts::conj(w)*hw;
105  norm_square.thread()+=std::pow(std::abs(w), 2.0);
106  }
107  num.put_thread_sum();
108  norm_square.put_thread_sum();
109  }

111  defs::ham_comp_t update_weights(const double &tau) {
112 #pragma omp parallel for default(none) shared(tau, stderr)
113  for (size_t irow_local = 0ul; irow_local < nrow_local(); ++irow_local) {
114  auto irow_walker_list = *m_local_subspace_list->m_irow(irow_local);
115  auto weight = m_walker_list->m_weight(irow_walker_list);
116  ASSERT(*weight==*weight)
117  auto h_weight = m_local_h_weights[irow_local];
118  ASSERT(h_weight==h_weight)
119  weight -= tau * h_weight;
120  }
