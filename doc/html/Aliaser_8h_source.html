30  void update(const defs::prob_t *probs, const size_t nprob) {
31  m_norm = std::accumulate(probs, probs+nprob, 0.0);
32  std::stack<size_t> smaller;
33  std::stack<size_t> larger;
34  for (size_t iprob = 0ul; iprob < m_nprob; ++iprob) {
35  m_prob_table[iprob] = probs[iprob] * m_nprob;
36  if (m_prob_table[iprob] < m_norm) smaller.push(iprob);
37  else larger.push(iprob);
38  }
39  size_t small, large;
40  while (!smaller.empty() && !larger.empty()) {
41  small = smaller.top();
42  smaller.pop();
43  large = larger.top();
44  larger.pop();
45  m_alias_table[small] = large;
46  m_prob_table[large] -= (m_norm - m_prob_table[small]);
47  if (m_prob_table[large] < m_norm) smaller.push(large);
48  else larger.push(large);
49  }
50  }

68  size_t draw() const {
69  size_t iprob = std::floor(m_prng.get().draw_float() * m_nprob);
70  ASSERT(iprob >= 0);
