14 /*
15  * notational conventions used are as follows:
16  * i, j: these refer to spin-orbitals if the integrals are spin-resolved,
17  * otherwise these are spatial orbitals
18  * ispin, jspin: these refer to the spin channel [0, 1] of the spin orbital,
19  * which is discarded in the non-spin-resolved case
20  * isat, jspat: these index the spatial orbital or kramers pair [0, nspatorb)
21  * flat_index: the position of the symmetrically unique integral value within m_data.
22  */
23 
24 template<typename T, size_t isym>
25 class Integrals_1e : public Integrals {
26  // none h
27  static_assert(isym == 1 || isym == 2, "Invalid symmetry parameter specified.");
28  const size_t m_nelem;
29  std::vector<T> m_data;
30 
31 public:
32  Integrals_1e(const size_t &norb, bool spin_resolved) :
33  Integrals(norb, spin_resolved), m_nelem(nelem(norb)) {
34  m_data.resize(m_nelem, 0.0);
35  }

37  Integrals_1e(std::string fname) :
38  Integrals_1e(FcidumpFileIterator<T>(fname).m_norb, FcidumpFileIterator<T>(fname).m_spin_resolved) {
39  FcidumpFileIterator<T> file_iterator(fname);
40  defs::inds inds(4);
41  T value;
42  while (file_iterator.next(inds, value)) {
43  if (valid_inds(inds)) {
44  set(inds[0], inds[1], value);
45  }
46  }
47  }

49  inline size_t flat_index(const size_t &i, const size_t &j) const {
50  if (isym == 1) return i + j * m_norb;
51  else return i <= j ? trig(i, j) : trig(j, i);
52  }
53 
54  void set(const size_t &i, const size_t &j, const T &value) {
55  auto iflat = flat_index(i, j);
56  auto conjd_value = (isym == 2 && i > j) ? consts::conj(value) : value;
57  if (consts::float_is_zero(m_data[iflat])) m_data[iflat] = conjd_value;
58  else {
59  ASSERT(consts::floats_nearly_equal(m_data[iflat], conjd_value));
60  }
61  }

63  void set(const size_t &ispat, const size_t &ispin,
64  const size_t &jspat, const size_t &jspin,
65  const T &value) {
66  set(spinorb(ispat, ispin), spinorb(jspat, jspin), value);
67  }

69  void set(const defs::inds &inds, const T &value) {
70  ASSERT(inds.size() == 4);
71  ASSERT(inds[2] == ~0ul && inds[3] == ~0ul);
72  set(inds[0], inds[1], value);
73  }

75  void set_from_fcidump(const defs::inds &inds, const T &value, bool spin_major = false) {
76  /*
77  * spin-resolved FCIDUMPs index in spinorbs, which may not may not be spin-major,
78  * depending on the program they were generated for. E.g. NECI uses spatial-major
79  * ordering throughout, so if the FCIDUMP supplied was intended for use with NECI,
80  * spin_major should be passed in as false.
81  */
82  if (!m_spin_resolved) set(inds, value);
83  else set(inds[0] / 2, inds[0] % 2, inds[1] / 2, inds[1] % 2, value);
84  }

86  T get(const size_t &i, const size_t &j) const {
87  //auto iflat = m_spin_resolved ? flat_index(i, j) : flat_index(i / 2, j / 2);
88  auto iflat = flat_index(i, j);
89  return (isym == 2 && i > j) ? consts::conj(m_data[iflat]) : m_data[iflat];
90  }

92  T get(const size_t &ispat, const size_t &ispin,
93  const size_t &jspat, const size_t &jspin) const {
94  return get(spinorb(ispat, ispin), spinorb(jspat, jspin));
95  }

98  T operator()(const size_t &i, const size_t &j) const {
99  /*
100  * return the one-body integral between the two SPINORS indexed by i and j
101  */
102  if (!m_spin_resolved && ((i<m_norb)!=(j<m_norb))) return 0.0;
103  auto iflat = m_spin_resolved ? flat_index(i, j) : flat_index(i % m_norb, j % m_norb);
104  return (isym == 2 && i > j) ? consts::conj(m_data[iflat]) : m_data[iflat];
105  }

107  static bool valid_inds(const defs::inds &inds) {
108  return inds[1] < (size_t) -1 && inds[2] == (size_t) -1;
109  }
110 
111  bool spin_conserving() const {
112  if (!m_spin_resolved) return true;
113  /*
114  * iterate through integrals looking for an example of a non-zero
115  * spin non-conserving one-body integral
116  */
117  for (size_t i = 0ul; i < m_norb; ++i) {
118  for (size_t j = 0ul; i < m_norb; ++i) {
119  if (!consts::float_is_zero(get(i, 0, j, 1))) return false;
120  }
121  }
122  return true;
123  }

125 private:
126  constexpr size_t nelem(const size_t &norb) {
127  static_assert(isym == 1 || isym == 2, "Invalid symmetry parameter specified.");
128  if (isym == 1) {
129  return norb * norb;
130  } else if (isym == 2) {
131  return trig(0, norb);
132  }
133  }
